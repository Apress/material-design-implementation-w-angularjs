               Following code mimics long running process.
                       angular.module("sampleApp", ["ngMaterial"])
           .controller('sampleController', function($scope, $timeout){
               $scope.progress=1;
               $scope.buffer=5;
               
               function updateProgress(){
                    $timeout(function(){
                        $scope.progress += 1;     // increment progress by 1.
                        $scope.buffer += 5; // increment buffer value (image a video file)
                        if($scope.progress<=100){ // Continue to call update progress recursively till progress is 100%
                            console.log($scope.progress);
                            updateProgress();
                              
                        }
                    }, 200);                   
               }
               
               updateProgress();

           });
