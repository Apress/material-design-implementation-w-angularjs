Form fields validations - samples
Fullname is mandatory
This field can not accept more than twenty characters
 
                    
        // Angular Material uses ngMessages for showing form validation failures. Make sure to add this module as a dependency
        angular.module('sampleApp', ['ngMaterial', 'ngMessages'])

        Template Code below,                     
        <!-- Encapsulate text field in a md-input-container element/directive for Material Design style text field -->
        <md-input-container flex>
            <!-- use required attribute for required field validation (HTML5)
                The validation alert will be shown in Angular Material way.
                
                Notice max length validation set with md-maxlength attribute/directive.
            -->
            <input type="text" placeholder="Enter full name" flex name="fullName" ng-model="fullName" required md-maxlength="20">
            
            <!-- $error object will populated with messages when form validations fail -->
            <div ng-messages="userApplication.fullName.$error">
                <!-- message for required field validation -->
                <div ng-message="required">Fullname is mandatory</div>
                <!-- message for max length validation -->
                <div ng-message="md-maxlength">This field can not accept more than twenty characters</div>
            </div>         
        </md-input-container>
Email address is mandatory
Invalid email address
Not a useful email. All on this form should end with mycompany.com
 
                    
        <md-input-container flex>
            <!-- use ng-pattern for regex validation.
                    Following has two types of validations for email
                    email validation checks basic email validation
                    pattern validation checks if the email from the expected company domain. 
                    multiple attribute allows all failing validation error messages shown at the same time.
                -->
            <input type="email" placeholder="Primary e-mail" flex name="email" ng-model="email" required ng-pattern="/^[a-z0-9]*@mycompany.com$/">
            <div ng-messages="userApplication.email.$error" multiple>
                <div ng-message="required">Email address is mandatory</div>
                <div ng-message="email">Invalid email address</div>
                <div ng-message="pattern">Not a useful email. All on this form should end with mycompany.com</div>
            </div>
        </md-input-container>

Phone number is mandatory
Invalid phone number
 
                
        <!--  use hint CSS class for showing a help hint text under the text field -->
        <md-input-container flex="40">
            <input type="text" placeholder="Phone number" flex name="phone" ng-model="phone" required ng-pattern="/^\+[0-9]{1,2} \([0-9]{3}\) [0-9]{3} [0-9]{4}$/">
            <div class="hint">+xx (yyy) zzz-zzzz; x= Country code; y= Regional code; z= Phone number </div>
            <div ng-messages="userApplication.phone.$error">
                <div ng-message="required">Phone number is mandatory</div>
                <div ng-message="pattern">Invalid phone number</div>
            </div>
        </md-input-container>
