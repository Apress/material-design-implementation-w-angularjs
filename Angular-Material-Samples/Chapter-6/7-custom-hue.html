Primary button Primary button (Hue-1) Primary button (Hue-2) Primary button (Hue-3) 
            

// Create a custom hue as follows         

angular.module("sampleApp", ["ngMaterial"])
        .config(function ($mdThemingProvider) {
                $mdThemingProvider
                        .theme('aCustomTheme')
                        .primaryPalette('cyan', {
                                'default': '900',
                                'hue-1': '50',
                                'hue-2': '200',
                                'hue-3': '600'
                        });
        });
