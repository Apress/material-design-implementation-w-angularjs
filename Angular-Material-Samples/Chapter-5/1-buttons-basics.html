Action Buttons Sample
Basic Buttons
Link Basic Raised Button 
                
                    <!-- Use md-button for a link. Use ng-href for the hyperlink -->
                    <md-button ng-href="https://material.angularjs.org/" target="__blank">Link</md-button>

                    <!-- A normal button. You will see the ink or ripple feedback on click of the button -->
                    <md-button ng-click="null">Basic</md-button>

                    <!-- Use md-raised class for raised effect -->
                    <md-button class="md-raised">Raised Button</md-button>

                    <!-- Icon button: Use CSS class md-icon-button. 
                        An Icon button needs ARIA label for accessibility reasons. 
                        Generally ngAria uses label for accessibility text. This is while using screen readers.
                        However, for an icon button, there is no label. Hence if we drop aria-label, it will result in a warning.
                    -->
                    <md-button ng-click="null" class="md-icon-button" aria-label="Sport">
                        <md-icon md-svg-src="images/ic_rowing_black_24px.svg"></md-icon>
                    </md-button>

Use md-primary CSS class for primary buttons

                    <!-- Icon button: Use CSS class md-icon-button. 
                        An Icon button needs ARIA label for accessibility reasons. 
                        Generally ngAria uses label for accessibility text. This is while using screen readers.
                        However, for an icon button, there is no label. Hence if we drop aria-label, it will result in a warning.

Disabled Buttons
Toggle to enable or disable buttons? Link Basic Raised Button 
                
                <!--
                    
use ng-disabled property for disabling and enabling a button. We can apply logic to do so on the fly.
 
                --> 
                                    
               <!-- Use ng-disabled=true/false to enable/disable a button 
                  Following sample has isButtonDisabled flag modified by a switch
                -->
                <span ng-init="isButtonDisabled=true"></span>
                <md-switch ng-model="isButtonDisabled">Toggle to enable or disable buttons?</md-switch>

                <!-- Use md-button for a link. Use ng-href for the hyperlink -->
                <md-button ng-disabled="isButtonDisabled" ng-href="https://material.angularjs.org/" target="__blank">Link</md-button>

                <!-- A normal button. You will see the ink or ripple feedback on click of the button -->
                <md-button ng-disabled="isButtonDisabled" ng-click="null">Basic</md-button>

                <!-- Use md-raised class for raised effect -->
                <md-button ng-disabled="isButtonDisabled" class="md-raised" md-no-ink>Raised Button</md-button>

                <!-- Icon button: Use CSS class md-icon-button. -->
                <md-button ng-disabled="isButtonDisabled" ng-click="null" class="md-icon-button" aria-label="Sport">
                    <md-icon md-svg-src="images/ic_lightbulb_outline_black_48px.svg"></md-icon>
                </md-button>

                <!-- Use class md-fab for Floating Action Button 
               
                -->
                <md-button ng-disabled="isButtonDisabled" ng-click="null" class="md-icon-button md-fab md-primary" aria-label="An Idea">
                    <md-icon md-svg-src="images/ic_lightbulb_outline_black_48px.svg"></md-icon>
                </md-button>

                <!-- md-mini for smaller FAB control -->
                <md-button ng-disabled="isButtonDisabled" ng-click="null" class="md-icon-button md-fab md-primary md-mini" aria-label="An Idea">
                    <md-icon md-svg-src="images/ic_lightbulb_outline_black_48px.svg"></md-icon>
