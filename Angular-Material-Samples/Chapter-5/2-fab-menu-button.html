FAB Menu Sample

 Control opening/closing the menu outside the FAB toolbar

Code Snippet

            
        <!--
            A logic outside the FAB Menu can control opening or clossing the component.
            Following is a switch, but it could be more sophisticated logic. When the a condition is true, toggle open/close the FAB Menu
        -->
        <md-switch ng-model="isOpen">Control opening/closing the menu outside the FAB toolbar</md-switch>
        
        
        <!-- 
                Use md-fab-toolbar element to create a fab menu 
                It decides position and the direction menu expands
            -->
        <md-fab-toolbar md-open="isOpen" class="md-fab-top-right" md-direction="left">
            <!--
                    The FAB button that triggers the menu is encapsulated in md-fab-trigger directive.
                -->
            <md-fab-trigger class="align-with-text">
                <md-button aria-label="Settings" class="md-fab md-primary">
                    <md-icon md-svg-src="images/ic_settings.svg"></md-icon>
                </md-button>
            </md-fab-trigger>

            <!--
                    The FAB toolbar options are in md-toolbar, same element used for a regular toolbar.
                -->
            <md-toolbar>
                <md-fab-actions class="md-toolbar-tools">
                    <md-button aria-label="Bluetooth Settings" class="md-icon-button">
                        <!-- Notice options created for FAB Menu are icon buttons -->
                        <md-icon md-svg-src="images/ic_settings_bluetooth.svg"></md-icon>
                    </md-button>
                    <md-button aria-label="Brightness Settings" class="md-icon-button">
                        <md-icon md-svg-src="images/ic_settings_brightness.svg"></md-icon>
                        </md-icon>
                    </md-button>
                    <md-button aria-label="Overscan Settings" class="md-icon-button">
                        <md-icon md-svg-src="images/ic_settings_overscan.svg"></md-icon>
                    </md-button>
                </md-fab-actions>
            </md-toolbar>
        </md-fab-toolbar>
