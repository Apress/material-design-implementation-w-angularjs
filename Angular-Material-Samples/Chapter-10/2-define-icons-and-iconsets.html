SVG Icons Sample

 
             
            // define icons and icon sets for consumption by md-icon directive
            angular.module("sampleApp", ["ngMaterial", "material.svgAssetsCache"])
            .config(function($mdIconProvider){
               $mdIconProvider
                .defaultIconSet('img/icons/sets/core-icons.svg', 24)
                .iconSet('communication','img/icons/sets/communication-icons.svg',24)
                .icon('account-circle','svg-icons/ic_account_circle.svg');
            });
            
            Usage below,
        <!-- Icon from default Icon Set -->
        <md-icon md-svg-icon="account-balance" style="width:48px; height:48px; color:red" ></md-icon>
        
        <!-- icon from communication icon set -->
        <md-icon md-svg-icon="communication:business" style="width:48px; height:48px; color:red" ></md-icon>
        
        <!-- Icon defined in the config function while bootstrapping the app -->
        <md-icon md-svg-icon='account-circle' style="width:48px; height:48px; color:red" ></md-icon>
